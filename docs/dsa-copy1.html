    for i in range(n):

def binary_search(arr, target):
    print(f"Searching for {target} in {arr}")
    low = 0
    high = len(arr) - 1
    while low <= high:
        mid = (low + high) // 2
        if arr[mid] == target:

            print(f"Found {target} at index {mid}")

            return mid
        elif arr[mid] < target:
            low = mid + 1
        else:
            high = mid - 1
    print(f"{target} not found.")
    return -1
