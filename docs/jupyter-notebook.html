def bubble_sort(arr):
    print("\nBubble Sort")
    n = len(arr)
    for i in range(n):
        for j in range(0, n - i - 1):

print(f"Compare {arr[j]} and {arr[j+1]}")

 if arr[j] > arr[j + 1]:
                print(f"Swap {arr[j]} <-> {arr[j+1]}")

arr[j], arr[j + 1] = arr[j + 1], arr[j]
        print(f"After pass {i+1}: {arr}")  
return arr

def insertion_sort(arr):
    print("\nInsertion Sort")
    for i in range(1, len(arr)):
        key = arr[i]
        j = i - 1

 print(f"Insert {key}")
        while j >= 0 and arr[j] > key:
            print(f"Move {arr[j]} to position {j+1}")
            arr[j + 1] = arr[j]
            j -= 1
        arr[j + 1] = key

print(f"Inserted {key}, Array now: {arr}")
    return arr

def linear_search(arr, target):
    print(f"\nLinear Search for {target}")
    for i in range(len(arr)):
        print(f"Check index {i}: {arr[i]}")
        if arr[i] == target:

print(f"Found {target} at index {i}")
            return i
    print("Not Found")
    return -1

def binary_search(arr, target):
    print(f"\nBinary Search for {target}")
    low, high = 0, len(arr) - 1
    while low <= high:
        mid = (low + high)

print(f"Check middle index {mid}: {arr[mid]}")
        if arr[mid] == target:
            print(f"Found {target} at index {mid}")
            return mid
        elif arr[mid] < target:

print(f"{target} > {arr[mid]} → go right")
            low = mid + 1
        else:
            print(f"{target} < {arr[mid]} → go left")
            high = mid - 1
    print("Not Found")
    return -1

def merge_sort(arr):
    print(f"\nMerge Sort on {arr}")
    if len(arr) > 1:
        mid = len(arr) // 2
        L = arr[:mid]
        R = arr[mid:]

print(f"Split into {L} and {R}")
        merge_sort(L)
        merge_sort(R)
        i = j = k = 0

while i < len(L) and j < len(R):
            print(f"Compare {L[i]} and {R[j]}")
            if L[i] < R[j]:
                arr[k] = L[i]
                i += 1
            else:
                arr[k] = R[j]
                j += 1
            k += 1

 while i < len(L):
            arr[k] = L[i]
            i += 1
            k += 1
        while j < len(R):
            arr[k] = R[j]
            j += 1
            k += 1

print(f"Merged: {arr}")

def quick_sort(arr):
    print(f"\nQuick Sort on {arr}")
    if len(arr) <= 1:
        return arr
    else:

 pivot = arr[0]
        print(f"Pivot: {pivot}")
        less = [x for x in arr[1:] if x <= pivot]
        greater = [x for x in arr[1:] if x > pivot]
        print(f"Less: {less}, Greater: {greater}")
        return quick_sort(less) + [pivot] + quick_sort(greate)
