import os
TASKS_FILE = "tasks.txt"

def load_tasks():
    print("Loading tasks from file...")

tasks = []
    if os.path.exists(TASKS_FILE):
        with open(TASKS_FILE, 'r') as f:
            lines = f.readlines()
            for line in lines:

status, task = line.strip().split("||")
                tasks.append({"task": task, "done": status == "1"})

print(f"Loaded {len(tasks)} tasks.")

 else:
        print("No existing task file found. Starting with an empty task list.")
    return tasks

def save_tasks(tasks):
    print("Saving tasks to file...")

 with open(TASKS_FILE, 'w') as f:
        for t in tasks:
            line = f"{'1' if t['done'] else '0'}||{t['task']}\n"
            f.write(line)

 print("Tasks saved successfully.")

def add_task(tasks):
    task = input("Enter a new task: ").strip()

if task:
        tasks.append({"task": task, "done": False})

 print(f"Task added: {task}")

  else:
        print("Empty task not added.")

def view_tasks(tasks):
    if not tasks:

print("No tasks to show.")
        return

print("\nYour Task List:")

for idx, t in enumerate(tasks):
        status = "✓ Done" if t["done"] else "⨯ Pending

print(f"{idx + 1}. {t['task']} - {status}")
    print(f"Total: {len(tasks)} task(s)")

def mark_complete(tasks):
    view_tasks(tasks)
    if not tasks:
        return
    try:

index = int(input("Enter task number to mark complete: ")) - 1
        if 0 <= index < len(tasks):
            tasks[index]["done"] = True

 print(f"Marked task '{tasks[index]['task']}' as complete.")

 else:
            print("Invalid task number.")

except ValueError:
        print("Invalid input. Please enter a number.")

def delete_task(tasks):
    view_tasks(tasks)
    if not tasks:
        return

 try:
        index = int(input("Enter task number to delete: ")) - 1
        if 0 <= index < len(tasks):
            deleted = tasks.pop(index)

print(f"Deleted task: {deleted['task']}")

else:
            print("Invalid task number.")

 except ValueError:
        print("Invalid input. Please enter a number.")

def main_menu():
    tasks = load_tasks()
    while True:

 print("\n=== Task Manager Menu ===")
        print("1. Add Task")
        print("2. View Tasks")
        print("3. Mark Task as Complete")
        print("4. Delete Task")
        print("5. Save & Exit")

 choice = input("Choose an option (1-5): ").strip()

if choice == '1':
            add_task(tasks)
        elif choice == '2':
            view_tasks(tasks)
        elif choice == '3':
            mark_complete(tasks)

elif choice == '4':
            delete_task(tasks)
        elif choice == '5':
            save_tasks(tasks)
            print("Goodbye. Tasks saved.")
            break

 else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main_menu()
